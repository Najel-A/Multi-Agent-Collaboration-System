  # pipelines/gold_pipeline.py
  from pyspark.sql import SparkSession
  from pyspark.sql.functions import col, when, lit, rand

  spark = SparkSession.builder.appName("gold_pipeline").getOrCreate()

  # ── Read silver parquet ──
  silver = spark.read.parquet("data/04-silver/")

  # ── 1a. SFT Splits (80/10/10) ──
  # Stratified by scenario_id so each split has all 19 scenarios
  train, val, test = silver.randomSplit([0.8, 0.1, 0.1], seed=42)

  train.write.mode("overwrite").parquet("data/05-gold/sft_train.parquet")
  val.write.mode("overwrite").parquet("data/05-gold/sft_val.parquet")
  test.write.mode("overwrite").parquet("data/05-gold/sft_test.parquet")

  # ── 1b. Classification Features (Model 2 Planner) ──
  features = silver.select(
      col("id"),
      col("fault.scenario_id").alias("label"),
      col("remediation.category").alias("category"),
      col("observations.metrics_snapshot.restarts").cast("int").alias("restarts"),
      col("observations.metrics_snapshot.oom_killed").cast("boolean").alias("oom_killed"),
      col("observations.kubectl_get_pods_len").alias("pods_len"),
      col("observations.kubectl_describe_pod_len").alias("describe_len"),
      col("observations.container_logs_len").alias("logs_len"),
      col("observations.kubectl_get_events_len").alias("events_len"),
      col("context.gitops_enabled").alias("has_gitops"),
      col("context.workload_kind").alias("workload_kind"),
      col("context.namespace").alias("namespace"),
  )
  features.write.mode("overwrite").parquet("data/05-gold/classification_features.parquet")

  # ── 1c. Debate Pairs (Model 4 Referee) ──
  # For each incident: correct diagnosis + wrong diagnosis from different scenario
  from pyspark.sql.window import Window
  from pyspark.sql.functions import row_number

  w = Window.orderBy(rand(seed=42))
  wrong_diags = silver.select(
      col("remediation.diagnosis").alias("wrong_diagnosis"),
      col("fault.scenario_id").alias("wrong_scenario"),
      row_number().over(w).alias("rn")
  )
  correct_diags = silver.select(
      col("id"),
      col("fault.scenario_id").alias("correct_scenario"),
      col("remediation.diagnosis").alias("correct_diagnosis"),
      col("observations"),
      row_number().over(w).alias("rn")
  )
  debate_pairs = correct_diags.join(wrong_diags, "rn").drop("rn")
  debate_pairs.write.mode("overwrite").parquet("data/05-gold/debate_pairs.parquet")

  print("Gold pipeline complete.")

  2a. Build Embeddings (Models 3, 4 — RAG + Memory)

  # pipelines/build_embeddings.py
  import json
  import faiss
  import numpy as np
  import pyarrow.parquet as pq
  from sentence_transformers import SentenceTransformer

  # Load silver data
  table = pq.read_table("data/05-gold/sft_train.parquet")
  df = table.to_pandas()

  # Concatenate signals into one text per incident
  def incident_text(row):
      obs = row.get("observations", {})
      return " ".join([
          obs.get("kubectl_get_pods", "") or "",
          obs.get("kubectl_get_events", "") or "",
          obs.get("kubectl_describe_pod", "") or "",
          obs.get("container_logs", "") or "",
      ])

  texts = [incident_text(row) for _, row in df.iterrows()]
  ids = df["id"].tolist()

  # Embed
  model = SentenceTransformer("all-MiniLM-L6-v2")  # 384-dim, fast, free
  vectors = model.encode(texts, show_progress_bar=True, batch_size=64)

  # Build FAISS index
  dim = vectors.shape[1]
  index = faiss.IndexFlatL2(dim)
  index.add(vectors.astype("float32"))

  # Save
  faiss.write_index(index, "data/06-embeddings/incident_vectors.faiss")

  # Save metadata
  import pandas as pd
  meta = pd.DataFrame({"id": ids, "scenario_id": df["fault"].apply(lambda f: f["scenario_id"])})
  meta.to_parquet("data/06-embeddings/metadata.parquet")

  print(f"Embedded {len(vectors)} incidents → FAISS index ({dim} dims)")

  2b. Fine-Tune Mistral 7B with LoRA (Model 3 Signal Executor)

  # pipelines/train_mistral_lora.py
  import torch
  from datasets import Dataset
  from transformers import (
      AutoModelForCausalLM, AutoTokenizer,
      TrainingArguments, BitsAndBytesConfig
  )
  from peft import LoraConfig, get_peft_model
  from trl import SFTTrainer
  import pyarrow.parquet as pq

  # ── Load training data from gold parquet ──
  table = pq.read_table("data/05-gold/sft_train.parquet")
  df = table.to_pandas()

  def build_prompt(row):
      ctx = row.get("context", {})
      obs = row.get("observations", {})
      rem = row.get("remediation", {})
      prompt = (
          f"Analyze these Kubernetes signals and diagnose the root cause.\n\n"
          f"CLUSTER: {ctx.get('cluster_id', '')}\n"
          f"NAMESPACE: {ctx.get('namespace', '')}\n"
          f"WORKLOAD: {ctx.get('workload_kind', '')}/{ctx.get('workload_name', '')}\n\n"
          f"=== kubectl get pods ===\n{obs.get('kubectl_get_pods', '')}\n\n"
          f"=== kubectl describe ===\n{obs.get('kubectl_describe_pod', '')}\n\n"
          f"=== events ===\n{obs.get('kubectl_get_events', '')}\n\n"
          f"=== logs ===\n{obs.get('container_logs', '')}\n\n"
          f"Respond with DIAGNOSIS and FIX_PLAN."
      )
      completion = (
          f"DIAGNOSIS: {rem.get('diagnosis', '')}\n"
          f"FIX_PLAN:\n" + "\n".join(f"{i+1}. {s}" for i, s in enumerate(rem.get('fix_plan', [])))
      )
      return {"text": f"<s>[INST] {prompt} [/INST] {completion}</s>"}

  records = [build_prompt(row) for _, row in df.iterrows()]
  dataset = Dataset.from_list(records)

  # ── Load Mistral 7B with 4-bit quantization (QLoRA) ──
  bnb_config = BitsAndBytesConfig(
      load_in_4bit=True,
      bnb_4bit_compute_dtype=torch.float16,
      bnb_4bit_quant_type="nf4",
  )

  model_id = "mistralai/Mistral-7B-v0.3"
  tokenizer = AutoTokenizer.from_pretrained(model_id)
  tokenizer.pad_token = tokenizer.eos_token

  model = AutoModelForCausalLM.from_pretrained(
      model_id,
      quantization_config=bnb_config,
      device_map="auto",
  )

  # ── LoRA config ──
  lora_config = LoraConfig(
      r=16,
      lora_alpha=32,
      target_modules=["q_proj", "k_proj", "v_proj", "o_proj"],
      lora_dropout=0.05,
      task_type="CAUSAL_LM",
  )

  model = get_peft_model(model, lora_config)
  model.print_trainable_parameters()
  # Trainable: ~4M / 7B (0.06%)

  # ── Train ──
  training_args = TrainingArguments(
      output_dir="agents/models/mistral_executor_lora",
      num_train_epochs=3,
      per_device_train_batch_size=4,
      gradient_accumulation_steps=4,
      learning_rate=2e-4,
      fp16=True,
      logging_steps=50,
      save_strategy="epoch",
      report_to="none",
  )

  trainer = SFTTrainer(
      model=model,
      train_dataset=dataset,
      tokenizer=tokenizer,
      args=training_args,
      dataset_text_field="text",
      max_seq_length=2048,
  )

  trainer.train()
  trainer.save_model("agents/models/mistral_executor_lora")
  print("Mistral LoRA fine-tune complete.")

  from dataclasses import dataclass, field
  from typing import Any

  @dataclass
  class AgentResult:
      agent: str
      status: str
      findings: dict[str, Any] = field(default_factory=dict)
      confidence: float = 0.0

  class BaseAgent:
      def __init__(self, name: str, model: Any = None):
          self.name = name
          self.model = model

      def run(self, incident: dict) -> AgentResult:
          raise NotImplementedError

  3b. Model 1: Single-Agent Baseline

  # agents/model1_baseline.py
  import openai

  class SREAgent(BaseAgent):
      """One agent. One call. No framework."""

      def __init__(self, model_id: str = "ft:gpt-4o-mini:..."):
          super().__init__("sre_agent")
          self.client = openai.OpenAI()
          self.model_id = model_id  # ← fine-tuned model from Step 2d

      def run(self, incident: dict) -> AgentResult:
          obs = incident["observations"]
          ctx = incident["context"]

          response = self.client.chat.completions.create(
              model=self.model_id,
              messages=[
                  {"role": "system", "content": "You are an on-call Kubernetes SRE."},
                  {"role": "user", "content": self._build_prompt(ctx, obs)}
              ]
          )

          return AgentResult(
              agent="sre_agent",
              status="success",
              findings=self._parse(response.choices[0].message.content),
              confidence=0.8,
          )

  3c. Model 2: Sequential Pipeline (CrewAI)

  # agents/model2_sequential.py
  from crewai import Agent, Task, Crew, Process

  def build_crew(fine_tuned_model_id: str):
      # ── Agents with fine-tuned models ──
      triage_planner = Agent(
          role="Triage Planner",
          goal="Classify the K8s incident category and severity",
          backstory="Senior SRE who triages pager alerts",
          llm=f"openai/{fine_tuned_model_id}",  # ← from Step 2d
      )

      diagnostic_executor = Agent(
          role="Diagnostic Executor",
          goal="Diagnose root cause and generate fix plan",
          backstory="Engineer who digs into logs and events",
          llm="gemini/gemini-1.5-flash",
      )

      evidence_reviewer = Agent(
          role="Evidence Reviewer",
          goal="Validate diagnosis against raw signals",
          backstory="Senior SRE who reviews incident reports",
          llm=f"openai/{fine_tuned_model_id}",
          allow_delegation=True,  # ← can send back to Executor
      )

      # ── Tasks ──
      classify = Task(
          description="Classify: {incident_signals}",
          expected_output="CATEGORY, SEVERITY, INVESTIGATION_FOCUS",
          agent=triage_planner,
      )

      diagnose = Task(
          description="Diagnose using category from previous task + full signals",
          expected_output="DIAGNOSIS, FIX_PLAN, COMMANDS, VERIFICATION, ROLLBACK",
          agent=diagnostic_executor,
          context=[classify],  # ← receives Planner's output
      )

      review = Task(
          description="Validate diagnosis against raw evidence",
          expected_output="VALID (true/false), CONFIDENCE, ISSUES",
          agent=evidence_reviewer,
          context=[diagnose],  # ← receives Executor's output
      )

      # ── Crew ──
      crew = Crew(
          agents=[triage_planner, diagnostic_executor, evidence_reviewer],
          tasks=[classify, diagnose, review],
          process=Process.sequential,
          verbose=True,
      )

      return crew

  3d. Model 3: Heterogeneous Collaboration (LangGraph)

  # agents/model3_heterogeneous.py
  from langgraph.graph import StateGraph, START, END
  from typing import TypedDict
  import faiss
  import openai
  from sentence_transformers import SentenceTransformer

  class IncidentState(TypedDict):
      incident: dict
      planner_diagnosis: str
      executor_diagnosis: str
      retrieved_incidents: list
      merged_result: dict
      approved: bool

  # ── Load trained models ──
  embedder = SentenceTransformer("all-MiniLM-L6-v2")
  faiss_index = faiss.read_index("data/06-embeddings/incident_vectors.faiss")
  oai_client = openai.OpenAI()

  # ── Load Mistral LoRA model ──
  from transformers import AutoModelForCausalLM, AutoTokenizer
  from peft import PeftModel

  base_model = AutoModelForCausalLM.from_pretrained("mistralai/Mistral-7B-v0.3", load_in_4bit=True)
  mistral_model = PeftModel.from_pretrained(base_model, "agents/models/mistral_executor_lora")
  mistral_tokenizer = AutoTokenizer.from_pretrained("mistralai/Mistral-7B-v0.3")

  # ── Agent functions ──
  def strategy_planner(state: IncidentState) -> IncidentState:
      """Branch A: classify + query RAG (GPT-4o-mini fine-tuned)"""
      incident = state["incident"]
      obs = incident["observations"]

      # Query FAISS for similar incidents
      text = obs.get("kubectl_get_pods", "") + " " + obs.get("kubectl_get_events", "")
      vec = embedder.encode([text])
      _, indices = faiss_index.search(vec.astype("float32"), k=5)
      state["retrieved_incidents"] = indices[0].tolist()

      # Call fine-tuned GPT-4o-mini
      response = oai_client.chat.completions.create(
          model="ft:gpt-4o-mini:...",  # ← from Step 2d
          messages=[{"role": "user", "content": f"Classify and diagnose:\n{text}"}]
      )
      state["planner_diagnosis"] = response.choices[0].message.content
      return state

  def signal_executor(state: IncidentState) -> IncidentState:
      """Branch B: independent analysis (Mistral 7B + LoRA, local)"""
      incident = state["incident"]
      obs = incident["observations"]

      prompt = f"[INST] Analyze signals:\n{obs} [/INST]"
      inputs = mistral_tokenizer(prompt, return_tensors="pt")
      outputs = mistral_model.generate(**inputs, max_new_tokens=512)
      state["executor_diagnosis"] = mistral_tokenizer.decode(outputs[0], skip_special_tokens=True)
      return state

  def consensus_merger(state: IncidentState) -> IncidentState:
      """Merge both diagnoses (Gemini Flash — large context)"""
      import google.generativeai as genai
      model = genai.GenerativeModel("gemini-1.5-flash")

      response = model.generate_content(
          f"Two agents analyzed the same K8s incident independently.\n\n"
          f"PLANNER: {state['planner_diagnosis']}\n\n"
          f"EXECUTOR: {state['executor_diagnosis']}\n\n"
          f"Merge: pick stronger evidence, resolve conflicts, produce final diagnosis."
      )
      state["merged_result"] = {"diagnosis": response.text}
      return state

  def safety_reviewer(state: IncidentState) -> IncidentState:
      """Check safety (GPT-4o-mini)"""
      response = oai_client.chat.completions.create(
          model="gpt-4o-mini",
          messages=[{"role": "user", "content": f"Is this fix safe?\n{state['merged_result']}"}]
      )
      state["approved"] = "approved" in response.choices[0].message.content.lower()
      return state

  # ── Build graph ──
  workflow = StateGraph(IncidentState)

  workflow.add_node("planner", strategy_planner)
  workflow.add_node("executor", signal_executor)
  workflow.add_node("merger", consensus_merger)
  workflow.add_node("safety", safety_reviewer)

  # Parallel branches
  workflow.add_edge(START, "planner")
  workflow.add_edge(START, "executor")

  # Both merge
  workflow.add_edge("planner", "merger")
  workflow.add_edge("executor", "merger")

  # Safety check with retry
  workflow.add_edge("merger", "safety")
  workflow.add_conditional_edges("safety", lambda s: END if s["approved"] else "merger")

  app = workflow.compile()

  # Usage:
  # result = app.invoke({"incident": incident_dict})
